        \foreach \i in {1,...,8}{%
          \ifnum\i=\insertsectionnumber%

          \else%
            \hyperlink{sec:\i}{\color{white}\i\ }%
          \fi%
        }%
      \end{beamercolorbox}%
    }%
    \vskip0pt%
    \fi%
  }
}
